[docs]    def is_ready(self, timeout=None):
        if timeout is None:
            timeout = self._check_timeout
        try:
            grpc.channel_ready_future(self._channel).result(timeout=timeout)
            return True
        except grpc.FutureTimeoutError:
            return False
